import Foundation

class ChatManager: ObservableObject, WebSocketServiceDelegate {
    @Published var chatRooms: [ChatRoom] = []

        let body = [

        guard let bodyData = try? JSONSerialization.data(withJSONObject: body) else {

            endpoint: "/api/chat-rooms",
            method: .POST,
            body: bodyData,
            headers: [:]
        )

            receiveCompletion: { (completion: Subscribers.Completion<Error>) in

                self.chatRooms.insert(chatRoom, at: 0)

    }

        guard let roomId = currentChatRoom?.id else { return }
        
        webSocketService.sendMessage(chatRoomId: roomId, content: content, type: type)
        
        // API í˜¸ì¶œë¡œë„ ì €ì¥
        let body = [

            endpoint: "/api/chat-rooms/\(roomId)/messages",
            method: .POST,
            body: bodyData,
            headers: [:]
        )
        .sink(
            receiveCompletion: { (completion: Subscribers.Completion<Error>) in
                if case .failure(let error) = completion {
                    print("ë©”ì‹œì§€ ì „ì†¡ ì‹¤íŒ¨: \(error)")

                // WebSocketìœ¼ë¡œ ì´ë¯¸ ì²˜ë¦¬ë¨
            }
        )
        .store(in: &cancellables)
    }
    
    func sendTyping(_ isTyping: Bool) {
        guard let roomId = currentChatRoom?.id else { return }
        webSocketService.sendTyping(chatRoomId: roomId, isTyping: isTyping)
    }
    
    func joinChatRoom(_ chatRoom: ChatRoom) {
        currentChatRoom = chatRoom
        webSocketService.joinRoom(chatRoom.id)
        loadMessages(for: chatRoom.id)
    }
    
    func leaveChatRoom() {
        if let roomId = currentChatRoom?.id {
            webSocketService.leaveRoom(roomId)
        }
        currentChatRoom = nil
        messages = []
        typingUsers = []
    }
    
    // MARK: - WebSocketServiceDelegate
    func webSocketDidConnect() {
        isConnected = true
    }
    
    func webSocketDidDisconnect() {
        isConnected = false
    }
    
    func webSocketDidReceiveMessage(_ message: Message) {
        if message.chatRoomId == currentChatRoom?.id {
            messages.append(message)
        }
        
        // ì±„íŒ…ë°© ëª©ë¡ ì—…ë°ì´íŠ¸
        if let index = chatRooms.firstIndex(where: { $0.id == message.chatRoomId }) {
            chatRooms[index].lastMessage = message
            chatRooms[index].updatedAt = message.createdAt
            
            // ìµœì‹  ë©”ì‹œì§€ë¥¼ ë§¨ ìœ„ë¡œ
            let room = chatRooms.remove(at: index)
            chatRooms.insert(room, at: 0)
        }
    }
    
    func webSocketDidReceiveTyping(userId: Int, user: User, isTyping: Bool) {
        if isTyping {
            if !typingUsers.contains(where: { $0.id == user.id }) {
                typingUsers.append(user)
            }
        } else {
            typingUsers.removeAll { $0.id == user.id }
        }
    }
}

// MARK: - Models
struct ChatRoom: Codable, Identifiable {
    let id: Int
    let name: String
    let isGroup: Bool
    let participants: [User]
    var lastMessage: Message?
    let createdAt: Date
    var updatedAt: Date
    
    var displayName: String {
        if isGroup {
            return name.isEmpty ? participants.map { $0.displayName }.joined(separator: ", ") : name
        } else {
            return participants.first?.displayName ?? "Unknown"
        }
    }
}

struct Message: Codable, Identifiable {
    let id: Int
    let chatRoomId: Int
    let senderId: String
    let content: String
    let messageType: String
    let createdAt: Date
    let sender: User
    let fileUrl: String?
    let fileName: String?
    let fileSize: Int?
    let voiceDuration: Double?
}

import Foundation
import UserNotifications
import UIKit
import Combine

class PushNotificationManager: NSObject, ObservableObject {
    static let shared = PushNotificationManager()
    
    @Published var isAuthorized = false
    @Published var deviceToken: String?
    
    private var cancellables = Set<AnyCancellable>()
    
    override init() {
        super.init()
        checkAuthorizationStatus()
    }
    
    func requestPermissions() {
        UNUserNotificationCenter.current().requestAuthorization(options: [.alert, .badge, .sound]) { granted, error in
            DispatchQueue.main.async {
                self.isAuthorized = granted
                if granted {
                    self.registerForRemoteNotifications()
                }
            }
        }
    }
    
    private func checkAuthorizationStatus() {
        UNUserNotificationCenter.current().getNotificationSettings { settings in
            DispatchQueue.main.async {
                self.isAuthorized = settings.authorizationStatus == .authorized
            }
        }
    }
    
    private func registerForRemoteNotifications() {
        DispatchQueue.main.async {
            UIApplication.shared.registerForRemoteNotifications()
        }
    }
    
    func didRegisterForRemoteNotifications(with deviceToken: Data) {
        let tokenParts = deviceToken.map { data in String(format: "%02.2hhx", data) }
        let token = tokenParts.joined()
        self.deviceToken = token
        
        // ì„œë²„ì— ë””ë°”ì´ìŠ¤ í† í° ì „ì†¡
        sendDeviceTokenToServer(token)
    }
    
    func didFailToRegisterForRemoteNotifications(with error: Error) {
        print("Failed to register for remote notifications: \(error)")
    }
    
    private func sendDeviceTokenToServer(_ token: String) {
        let body = [
            "deviceToken": token,
            "platform": "ios"
        ]
        
        guard let bodyData = try? JSONSerialization.data(withJSONObject: body) else {
            return
        }
        
        APIService.shared.request<TokenResponse>(
            endpoint: "/api/push-tokens",
            method: .POST,
            body: bodyData,
            headers: [:]
        )
        .sink(
            receiveCompletion: { (completion: Subscribers.Completion<Error>) in
                if case .failure(let error) = completion {
                    print("ë””ë°”ì´ìŠ¤ í† í° ì „ì†¡ ì‹¤íŒ¨: \(error)")
                }
            },
            receiveValue: { (response: TokenResponse) in
                print("ë””ë°”ì´ìŠ¤ í† í° ë“±ë¡ ì„±ê³µ")
            }
        )
        .store(in: &cancellables)
    }
    
    func handleNotification(_ userInfo: [AnyHashable: Any]) {
        // í‘¸ì‹œ ì•Œë¦¼ ì²˜ë¦¬ ë¡œì§
        if let chatRoomId = userInfo["chatRoomId"] as? Int {
            // íŠ¹ì • ì±„íŒ…ë°©ìœ¼ë¡œ ì´ë™
            NotificationCenter.default.post(
                name: NSNotification.Name("OpenChatRoom"),
                object: nil,
                userInfo: ["chatRoomId": chatRoomId]
            )
        }
    }
    
    func setBadgeCount(_ count: Int) {
        DispatchQueue.main.async {
            UIApplication.shared.applicationIconBadgeNumber = count
        }
    }
    
    func clearBadge() {
        setBadgeCount(0)
    }
}
